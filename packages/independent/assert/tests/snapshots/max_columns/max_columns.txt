# at property value
actual: {
  foo: …"fghijklmn",
}
expect: {
  foo: …"FGHIJKLMN",
}

# at property key
actual: {
  …"roperty key that will be truncated",
}
expect: {
  …"roperty key that will be truncated",
}

# at property name last char
actual: {
  …cdefgh,
}
expect: {
  …cdefgh,
}

# at property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at space after property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value first char
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value second char
actual: {
  abcdefgh: …,
}
expect: {
  abcdefgh: …,
}

# at property value second char (and value width is 1)
actual: {
  abcdefgh: 0,
}
expect: {
  abcdefgh: 1,
}

# at property value third char
actual: {
  abcdefgh: t…,
}
expect: {
  abcdefgh: f…,
}
