# string open quote
actual: …
expect: …

# string first char
actual: …
expect: …

# string second char
actual: …
expect: …

# string third char
actual: "a"…
expect: "A"…

# string fourth char
actual: "ab"…
expect: "AB"…

# string last char
actual: "abcdefgh"…
expect: "ABCDEFGH"…

# string close quote
actual: "abcdefghij"
expect: "ABCDEFGHIJ"

# at property value
actual: {
  zzz: "abcdefghi"…,
}
expect: {
  zzz: "ABCDEFGHI"…,
}

# at property key
actual: {
  …"roperty key that will be truncated",
}
expect: {
  …"roperty key that will be truncated",
}

# at property name last char
actual: {
  …cdefgh,
}
expect: {
  …cdefgh,
}

# at property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at space after property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value first char
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value second char
actual: {
  abcdefgh: …,
}
expect: {
  abcdefgh: …,
}

# at property value second char (and value width is 1)
actual: {
  abcdefgh: 0,
}
expect: {
  abcdefgh: 1,
}

# at property value third char
actual: {
  abcdefgh: t…,
}
expect: {
  abcdefgh: f…,
}
