# truncate property value
actual: {
  foo: "abcdefghij…,
}
expect: {
  foo: "ABCDEFGHIJ…,
}

# truncate property key
actual: {
  "a quite long property key that will…,
}
expect: {
  "a quite long property key that will…,
}

# truncate right after property name
actual: {
  abcdefghijkl,
}
expect: {
  abcdefghijkl,
}

# truncate right after property separator
actual: {
  abcdefghijkl,
}
expect: {
  abcdefghijkl,
}

# truncate exactly on first value column
actual: {
  abcdefghijkl,
}
expect: {
  abcdefghijkl,
}
