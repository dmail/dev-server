# at property value
actual: {
  foo: "abcdefghij…,
}
expect: {
  foo: "ABCDEFGHIJ…,
}

# at property key
actual: {
  "a quite long property key that will…,
}
expect: {
  "a quite long property key that will…,
}

# at property name last char
actual: {
  abcdef…,
}
expect: {
  abcdef…,
}

# at property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at space after property name separator
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value first char
actual: {
  abcdefgh,
}
expect: {
  abcdefgh,
}

# at property value second char
actual: {
  abcdefgh: …,
}
expect: {
  abcdefgh: …,
}

# at property value third char
actual: {
  abcdefgh: t…,
}
expect: {
  abcdefgh: f…,
}
