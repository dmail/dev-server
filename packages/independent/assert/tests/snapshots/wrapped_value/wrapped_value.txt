# 10 vs valueOf(10)
actual: 10
expect: {
  valueOf(): 10,
}

# valueOf(10) vs 10
actual: {
  valueOf(): 10,
}
expect: 10

# valueOf(10) vs valueOf(11)
actual: {
  valueOf(): 10,
}
expect: {
  valueOf(): 11,
}

# valueOf(10) vs valueOf(10)
actual: {
  a: true,
  valueOf(): 10,
}
expect: {
  b: false,
  valueOf(): 10,
}

# valueOf({ a: true }) vs { a: true }
actual: {
  valueOf(): {
    a: true,
  },
}
expect: {
  a: false,
}

# valueOf with object tag
actual: Signal(10)
expect: false

# Symbol.toPrimitive vs primitive
actual: {
  [Symbol.toPrimitive()]: "10",
}
expect: "10"

# primitive vs Symbol.toPrimitive
actual: "10"
expect: {
  [Symbol.toPrimitive()]: "10",
}
